use tree_sitter::{Query, QueryCursor, Node, StreamingIterator};
use cgraph_core::{SymbolNode, Language, SymbolKind};
use crate::classify::classify_function;

/// Extract all exported symbols from the AST root node.
/// Uses the pre-compiled symbol_query to match export_statement patterns.
/// Also extracts non-exported top-level function/variable declarations for call edge resolution.
pub fn extract_symbols(
    root: Node,
    source: &str,
    file_path: &str,
    language: Language,
    symbol_query: &Query,
) -> Vec<SymbolNode> {
    let mut nodes = Vec::new();
    let mut cursor = QueryCursor::new();
    let mut matches = cursor.matches(symbol_query, root, source.as_bytes());

    let name_idx = symbol_query.capture_index_for_name("symbol_name");
    let default_name_idx = symbol_query.capture_index_for_name("default_name");

    while let Some(m) = matches.next() {
        let pattern_idx = m.pattern_index;

        // Map pattern_index to SymbolKind
        let base_kind = match pattern_idx {
            0 => SymbolKind::Function,    // exported function declaration
            1 => SymbolKind::Function,    // exported arrow function
            2 => SymbolKind::Interface,   // exported interface
            3 => SymbolKind::Type,        // exported type alias
            4 => SymbolKind::Class,       // exported class
            5 => SymbolKind::Enum,        // exported enum
            _ => continue,  // skip default export patterns for now, handle separately
        };

        // Find the @symbol_name capture
        let capture_idx = match name_idx {
            Some(idx) => idx,
            None => continue,
        };

        for cap in m.captures {
            if cap.index == capture_idx {
                let name = &source[cap.node.byte_range()];
                let node = cap.node;

                // Reclassify functions as hooks if they match use* convention
                let kind = if matches!(base_kind, SymbolKind::Function) {
                    classify_function(name)
                } else {
                    base_kind.clone()
                };

                nodes.push(SymbolNode {
                    id: format!("{}::{}", file_path, name),
                    name: name.to_string(),
                    kind,
                    file_path: file_path.to_string(),
                    language: language.clone(),
                    line_start: node.start_position().row as u32 + 1,
                    line_end: node.end_position().row as u32 + 1,
                    is_exported: true,
                });
                break; // one symbol per match
            }
        }
    }

    // Also extract non-exported top-level functions for intra-file call edges
    // Use a separate tree walk for non-exported declarations
    extract_non_exported_functions(root, source, file_path, language.clone(), &mut nodes);

    nodes
}

/// Extract non-exported top-level function declarations.
/// These are needed so that intra-file call edges can reference them.
fn extract_non_exported_functions(
    root: Node,
    source: &str,
    file_path: &str,
    language: Language,
    nodes: &mut Vec<SymbolNode>,
) {
    let mut tree_cursor = root.walk();
    // Walk top-level children only
    if tree_cursor.goto_first_child() {
        loop {
            let node = tree_cursor.node();
            // Skip export_statements (already captured above)
            if node.kind() == "function_declaration" {
                if let Some(name_node) = node.child_by_field_name("name") {
                    let name = &source[name_node.byte_range()];
                    // Check it's not already in nodes (exported ones already captured)
                    let already_exists = nodes.iter().any(|n| n.name == name);
                    if !already_exists {
                        let kind = classify_function(name);
                        nodes.push(SymbolNode {
                            id: format!("{}::{}", file_path, name),
                            name: name.to_string(),
                            kind,
                            file_path: file_path.to_string(),
                            language: language.clone(),
                            line_start: node.start_position().row as u32 + 1,
                            line_end: node.end_position().row as u32 + 1,
                            is_exported: false,
                        });
                    }
                }
            } else if node.kind() == "lexical_declaration" {
                // Non-exported const arrow functions
                let mut decl_cursor = node.walk();
                if decl_cursor.goto_first_child() {
                    loop {
                        let child = decl_cursor.node();
                        if child.kind() == "variable_declarator" {
                            if let Some(name_node) = child.child_by_field_name("name") {
                                if let Some(value_node) = child.child_by_field_name("value") {
                                    if value_node.kind() == "arrow_function" {
                                        let name = &source[name_node.byte_range()];
                                        let already_exists = nodes.iter().any(|n| n.name == name);
                                        if !already_exists {
                                            let kind = classify_function(name);
                                            nodes.push(SymbolNode {
                                                id: format!("{}::{}", file_path, name),
                                                name: name.to_string(),
                                                kind,
                                                file_path: file_path.to_string(),
                                                language: language.clone(),
                                                line_start: child.start_position().row as u32 + 1,
                                                line_end: child.end_position().row as u32 + 1,
                                                is_exported: false,
                                            });
                                        }
                                    }
                                }
                            }
                        }
                        if !decl_cursor.goto_next_sibling() {
                            break;
                        }
                    }
                }
            }
            if !tree_cursor.goto_next_sibling() {
                break;
            }
        }
    }
}

use cgraph_core::SymbolKind;

/// Reclassify a Function as Hook if it follows React hook naming convention.
/// Hook: starts with "use" and 4th character is uppercase (per D-32).
/// Examples: useCurrentUser -> Hook, useState -> Hook, useful -> Function, use -> Function
pub fn classify_function(name: &str) -> SymbolKind {
    if name.starts_with("use")
        && name.len() > 3
        && name.chars().nth(3).is_some_and(|c| c.is_uppercase())
    {
        SymbolKind::Hook
    } else {
        SymbolKind::Function
    }
}

#[cfg(test)]
mod tests {

        assert_eq!(classify_function("useCurrentUser"), SymbolKind::Hook);
        assert_eq!(classify_function("useState"), SymbolKind::Hook);
        assert_eq!(classify_function("useEffect"), SymbolKind::Hook);
    }

    #[test]
    fn non_hook_functions() {
        assert_eq!(classify_function("fetchUser"), SymbolKind::Function);
        assert_eq!(classify_function("useful"), SymbolKind::Function);
        assert_eq!(classify_function("use"), SymbolKind::Function);
        assert_eq!(classify_function("user"), SymbolKind::Function);

}
